        % Value changed function: PredictButton
        function PredictButtonValueChanged(app, event)
            value = app.PredictButton.Value;
            switch value
                case 1
                    %ALL CODE
                    app.areafinal.Visible = true;
                    app.areadet.Visible = true;
                    app.xedit.Enable = false;
                    app.yedit.Enable = false;
                    app.mdd.Enable = false;
                    app.ddd.Enable = false;
                    app.ffmcedit.Enable = false;
                    app.dmcedit.Enable = false;
                    app.dcedit.Enable = false;
                    app.isiedit.Enable = false;
                    app.tempedit.Enable = false;
                    app.rhedit.Enable = false;
                    app.windedit.Enable = false;
                    app.rainedit.Enable = false;
                    app.PredictButton.Text = "New Sheet"
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                    %COORDINATES
                        X = str2double(app.xedit.Value);
                        Y = str2double(app.yedit.Value);
                    %PERIOD DATE
                        month = "jan" ;
                        day = "sun" ;
                        switch(app.mdd.Value)
                            case 'jan'
                                month = "jan";
                            case 'feb'
                                month = "feb";
                            case 'mar'
                                month = "mar";
                            case 'apr'
                                month = "apr";
                            case 'may'
                                month = "may";
                            case 'jun'
                                month = "jun";
                            case 'jul'
                                month = "jul";
                            case 'aug'
                                month = "aug";
                            case 'sep'
                                month = "sep";
                            case 'oct'
                                month = "oct";
                            case 'nov'
                                month = "nov";
                            case 'dec'
                                month = "dec";
                            otherwise
                                month = "jan";
                        end
                        
                        switch(app.ddd.Value)
                            case 'sun'
                                day = "sun";
                            case 'mon'
                                day = "mon";
                            case 'tue'
                                day = "tue";
                            case 'wed'
                                day = "wed";
                            case 'thu'
                                day = "thu";
                            case 'fri'
                                day = "fri";
                            case 'sat'
                                day = "sat";
                            otherwise
                                day = "sun";
                        end
                       
                    %SOME FACTORS
                        FFMC = str2double(app.ffmcedit.Value);
                        DMC = str2double(app.dmcedit.Value);
                        DC = str2double(app.dcedit.Value);
                        ISI = str2double(app.isiedit.Value);
                    %ENVIRONMENT
                        temp = str2double(app.tempedit.Value);
                        RH = str2double(app.rhedit.Value);
                        wind = str2double(app.windedit.Value);
                        rain = str2double(app.rainedit.Value);
                    %RESULTS
                    area = 0;
                    Fire = 0;
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                  
                    
                    load 'matlab2.mat';
                    T2 = table(X, Y, month, day, FFMC, DMC, DC, ISI, temp, RH, wind, rain, Fire);
                    yfit2 = RFClass.predictFcn(T2) 
                    
                    
                    load 'matlab.mat';
                    T = table(X, Y, month, day, FFMC, DMC, DC, ISI, temp, RH, wind, rain, area);
                    yfit = RF.predictFcn(T)
                    
                    
                            
                    
                    if ~yfit2
                        app.areafinal.Value = '0'
                    else
                       yy = sprintf('%.4f',yfit)
                       app.areafinal.Value = yy;
                       X =int16(X)
                       Y =int16(Y) 
                       yfit5 = int16(yfit);
                       plot(app.UIAxes,X, Y, '.r', 'MarkerSize',yfit5);
            
                       
                    end
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
                case 0
                    %
                    app.areafinal.Value = '?'
                    app.xedit.Enable = true;
                    app.yedit.Enable = true;
                    app.mdd.Enable = true;
                    app.ddd.Enable = true;
                    app.ffmcedit.Enable = true;
                    app.dmcedit.Enable = true;
                    app.dcedit.Enable = true;
                    app.isiedit.Enable = true;
                    app.tempedit.Enable = true;
                    app.rhedit.Enable = true;
                    app.windedit.Enable = true;
                    app.rainedit.Enable = true;
                    app.PredictButton.Text = "Predict"
                case 0
            end
        end

            end

            % Delete UIFigure when app is deleted
